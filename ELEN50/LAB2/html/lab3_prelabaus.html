Vs = 5;
Rs = 10e4;
Rvm = 10e5;
R1 = [15e3;20e3;30e3;56e3;100e3];

% Expected Value of v1
I = Vs./(Rs+R1);
V1 = Is .* R1;

% Expected value of v1L
R1vm = (R1.*Rvm)./(Rvm+R1);

I = Vs./Rt;
